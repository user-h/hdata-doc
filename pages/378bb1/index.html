    快捷键

    # Sublime快捷键

    # 列编辑
Ctrl + Shift + L

# 大小写转换
Ctrl + K + U   # 大写
Ctrl + K + L   # 小写

# 命令行
Ctrl + Shift + P

# 快速查找 匹配括号
Ctrl+R         # 快速查找函数
Ctrl+M         # 跳转到另一半括号
Ctrl+Shift+M   # 快速匹配括号内容
Ctrl+Shift+[   # 折叠代码
Ctrl+Shift+]   # 展开代码
Alt+-          # 回到上一次编辑的地方，多次按可以追溯到编辑轨迹
Shift+Alt+-    # 对应的返回下一编辑处，直到最终编辑的地方。这里说的编辑处，实际上就是指的光标所在的地方
Ctrl+;         # 查找变量名（也会显示函数名称）

# 文件
["ctrl+shift+n"]		# 新建窗口
["ctrl+shift+w"]		# 关闭窗口
["ctrl+o"]		        # 提示打开文件
["ctrl+shift+t"]		# 重新打开最后关闭的文件
["ctrl+n"]		        # 新建文件
["ctrl+s"]		        # 保存文件
["ctrl+shift+s"]		# 另存为
["ctrl+w"]		        # 关闭文件

# 窗口内
["ctrl+k", "ctrl+b"]		# 侧边栏
["f11"]		                # 切换全屏
["shift+f11"]               # toggle_distraction_free

    # IDEA快捷键

    # 常用快捷键
Ctrl + X          # 删除行
Ctrl + D          # 复制行
Ctrl + /          # 或 Ctrl+Shift+/ 注释（// 或者/…/ ）
Ctrl + E          # 最近打开的文件
Ctrl + H          # 显示类结构图
Ctrl + F          # 搜索
Ctrl + N          # 查找类
Ctrl + G        # 跳转到某一行
Alt + 1           # 快速打开或隐藏工程面板
Alt +  left/right # 切换打开的文件视图
Alt +  Up/Down    # 在方法间快速移动定位
Alt + 回车         # 导入包,自动修正
ctrl + h        # 查找实现?

代码补全	
Ctrl + Alt + V   # 自动补全变量名及类型。例如把括号内的SQL赋成一个变量 点击setting勾选Local definition可以自动补全变量类型
