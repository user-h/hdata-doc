## 函数和类

```python
# ====================函数=============================
def run():
    print('Car started...')
    
    
def stop():
    print('Car stopped...')
    
    
def error():
    print("I can't understand...")


def message(status):
    if status in 'stop':
        print('Car is already stopped')
    else:
        print('Car is already started')


status = 'stop'
i = 6
while i>=1:
    i-=1
    order = input('>')
    if order in 'start':
        if status in 'stop':
            run()
            status = 'start'
        else:
            message(status)
    elif order in 'stop':
        if status in 'start':
            stop()
            status = 'stop'
        else:
            message(status)
    else:
        error()
else:
    print('Finish')
    
# ====================类===========================
class Mammal:
    def __init__(self,name):
        self.name = name
    
    def walk(self):
        print(f'{self.name} walking')
        
    def shout(self):
        print(f'{self.name} shouting')
        
        
class Dog(Mammal):
    print('===============Dog===============')
    def shout(self):
        print(f'{self.name} 汪汪')

class Cat(Mammal):
    print('===============Cat===============')
    pass


mammal = Mammal('哺乳动物')
mammal.walk()
mammal.shout()

cat = Cat('喵喵')
cat.walk()
cat.shout()

dog = Dog('二哈')
dog.walk()
dog.shout()
```
