## Oracle

### 一个字符串包含另一个字符串中的所有字符
```sql
create or replace function checks(v_a varchar2,v_b varchar)
return number
as
  num number;
  cou number;
begin
  num := -1;
  cou:=0;
  for i in 1..length(v_b) loop
     if instr(v_a,substr(v_b,i,1))>0 then
      cou:=cou+1;
     end if;
  end loop;
  if cou=length(v_b) then
  return cou;
  end if;
  --dbms_output.put_line(cou||'    '||length(v_b));
return num;
end;
```
