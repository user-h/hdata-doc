## Oracle

```sql
-- 一个字符串包含另一个字符串中的所有字符
create or replace function checks(v_a varchar2,v_b varchar)
return number
as
  num number;
  cou number;
  j number;
  index1 number;
begin
  num := -1;
  cou:=0;
  j := 0;
  index1 := 1;
  for i in 1..length(v_b) loop
     j := instr( substr(v_a,index1,length(v_a)), substr(v_b,i,1) );
     if j > 0 then
      cou:=cou+1;
      index1 := index1 + j;
     else
      return num;
     end if;
  end loop;

end;
```
